Custom Checkboxes

To use custom checkboxes, be sure to include your input before your label. Failing to do so will break styles.

<!-- Custom Checkboxes -->
<fieldset class="column column--full">
  <label>Checkboxes</label>

    <label for="dashing-checkbox1--custom" class="inline">Option 1</label>

Checkbox Cards

To use custom checkboxes, be sure to include your input before your label. Failing to do so will break styles.

To use smaller checkbox cards apply the .small class to the parent div.

<!-- Checkbox Cards -->
<fieldset class="column column--full">
	<label>Checkbox Card</label>
	<div class="checkbox-card">
		<input type="checkbox" name="checkbox-test" id="checkbox-test1" checked/>
		<label for="checkbox-test1" class="card">Option 1</label>
	</div>
	<div class="checkbox-card">
		<input type="checkbox" name="checkbox-test" id="checkbox-test2" />
		<label for="checkbox-test2" class="card">Option 2</label>
	</div>
	<div class="checkbox-card">
		<input type="checkbox" name="checkbox-test" id="checkbox-test3" disabled />
		<label for="checkbox-test3" class="card disabled">Disabled</label>
	</div>
</fieldset>

<!-- Small Checkbox Cards -->
<fieldset class="column column--full">
	<label>Small Checkbox Card</label>
	<div class="checkbox-card small">
		<input type="checkbox" name="checkbox-test" id="checkbox-test1" checked/>
		<label for="checkbox-test1" class="card">Option 1</label>
	</div>
	<div class="checkbox-card small">
		<input type="checkbox" name="checkbox-test" id="checkbox-test2" />
		<label for="checkbox-test2" class="card">Option 2</label>
	</div>
	<div class="checkbox-card small">
		<input type="checkbox" name="checkbox-test" id="checkbox-test3" disabled />
		<label for="checkbox-test3" class="card disabled">Disabled</label>
	</div>
</fieldset>

//Include these variables in your theme file to change the color or padding of your checkbox
$card-checkbox--active: $blue !default; //Color of checkbox and card when checked
$card-checkbox--focus: $blue-300 !default; //Color of border around card-checkbox when focused
$card-checkbox--padding: 1rem !default; //Padding of checkbox
